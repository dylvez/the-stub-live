```typescript
import { ref, uploadBytes, getDownloadURL } from 'firebase/storage';
import { storage } from './config';
import type { StubPhoto } from '@/types';

const MAX_WIDTH = 1920;
const JPEG_QUALITY = 0.8;
const MAX_FILE_SIZE = 10 * 1024 * 1024; // 10MB
const MAX_PHOTOS = 10;
const ALLOWED_TYPES = ['image/jpeg', 'image/png', 'image/webp'];

export { MAX_PHOTOS, MAX_FILE_SIZE, ALLOWED_TYPES };

/** Compress an image file to JPEG, max width 1920px */
export function compressImage(file: File): Promise<Blob> {
  return new Promise((resolve, reject) => {
    const img = new Image();
    const url = URL.createObjectURL(file);

    img.onload = () => {
      URL.revokeObjectURL(url);

      let { width, height } = img;
      if (width > MAX_WIDTH) {
        height = Math.round((height * MAX_WIDTH) / width);
        width = MAX_WIDTH;
      }

      const canvas = document.createElement('canvas');
      canvas.width = width;
      canvas.height = height;

      const ctx = canvas.getContext('2d');
      if (!ctx) {
        reject(new Error('Canvas not supported'));
        return;
      }

      ctx.drawImage(img, 0, 0, width, height);
      canvas.toBlob(
        (blob) => {
          if (blob) resolve(blob);
          else reject(new Error('Compression failed'));
        },
        'image/jpeg',
        JPEG_QUALITY,
      );
    };

    img.onerror = () => {
      URL.revokeObjectURL(url);
      reject(new Error('Failed to load image'));
    };

    img.src = url;
  });
}

export interface UploadResult {
  photos: StubPhoto[];
  failedCount: number;
}

/** Upload photos to Cloud Storage and return StubPhoto array + failure count */
export async function uploadStubPhotos(
  stubId: string,
  files: File[],
): Promise<UploadResult> {
  const photos: StubPhoto[] = [];
  let failedCount = 0;

  for (const file of files) {
    const photoId = `${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;
    const storagePath = `stubs/${stubId}/${photoId}.jpg`;
    const storageRefObj = ref(storage, storagePath);

    try {
      const compressed = await compressImage(file);
      await uploadBytes(storageRefObj, compressed, { contentType: 'image/jpeg' });
      const url = await getDownloadURL(storageRefObj);

      photos.push({ url, storageRef: storagePath });
    } catch (err) {
      // Retry once
      try {
        const compressed = await compressImage(file);
        await uploadBytes(storageRefObj, compressed, { contentType: 'image/jpeg' });
        const url = await getDownloadURL(storageRefObj);
        photos.push({ url, storageRef: storagePath });
      } catch {
        failedCount++;
        console.warn(`Failed to upload photo ${file.name}:`, err);
      }
    }
  }

  return { photos, failedCount };
}
```

  if (params.lastfmBio) parts.push(`Bio excerpt: ${params.lastfmBio.slice(0, 500)}`);
  return parts.join("\n");
}

exports.generateArtistBriefing = onCall(
  {
    region: "us-east1",
    secrets: [anthropicKey],
    timeoutSeconds: 60,
    memory: "256MiB",
  },
  async (request) => {
    const { artistId, name, genres, tags, spotifyPopularity, lastfmBio, topTrackNames, listenerCount } = request.data;

    if (!artistId || !name) {
      throw new HttpsError("invalid-argument", "artistId and name are required");
    }

    const db = getFirestore();
    const briefingRef = db.collection("briefings").doc(artistId);

    // Check cache
    const existing = await briefingRef.get();
    if (existing.exists) {
      const data = existing.data();
      const age = Date.now() - data.generatedAt.toMillis();
      const thirtyDays = 30 * 24 * 60 * 60 * 1000;

      if (age < thirtyDays && data.briefingText && !data.generating) {
        return { briefing: JSON.parse(data.briefingText), cached: true };
      }

      // Check lock — stale after 5 minutes
      if (data.generating && age < 5 * 60 * 1000) {
        return { status: "generating" };
      }
    }

    // Set generating lock
    await briefingRef.set(
      { generating: true, generatedAt: FieldValue.serverTimestamp() },
      { merge: true },
    );

    try {
      const userPrompt = buildUserPrompt({ name, genres, tags, spotifyPopularity, lastfmBio, topTrackNames, listenerCount });

      const body = JSON.stringify({
        model: "claude-sonnet-4-20250514",
        max_tokens: 512,
        temperature: 0.7,
        system: SYSTEM_PROMPT,
        messages: [{ role: "user", content: userPrompt }],
      });

      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-api-key": anthropicKey.value(),
          "anthropic-version": "2023-06-01",
        },
        body,
      });

      if (!response.ok) {
        const errText = await response.text();
        throw new Error(`Claude API error ${response.status}: ${errText}`);
      }

      const result = await response.json();
      let jsonStr = result.content?.[0]?.text?.trim() ?? "";
      if (jsonStr.startsWith("```")) {
        jsonStr = jsonStr.replace(/^```(?:json)?\n?/, "").replace(/\n?```$/, "");
      }

      const briefing = JSON.parse(jsonStr);

      // Save to Firestore
      await briefingRef.set({
        artistId,
        artistName: name,
        briefingText: JSON.stringify(briefing),
        generatedAt: FieldValue.serverTimestamp(),
        generating: false,
      });

      return { briefing, cached: false };
    } catch (err) {
      // Clear lock on failure
      await briefingRef.update({ generating: false }).catch(() => {});
      console.error("Briefing generation failed:", err);
      throw new HttpsError("internal", "Failed to generate briefing");
    }
  },
);
